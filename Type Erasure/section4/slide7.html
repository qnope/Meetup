Step 5 — Copy Semantics

unique_ptr is move-only. We need clone() for deep copy. 


struct Concept {
    virtual ~Concept() = default;
    virtual void draw() const = 0;
    virtual std::unique_ptr<Concept>
        clone() const = 0;
};

template <typename T>
struct Model : Concept {
    T obj;
    Model(T obj) : obj(std::move(obj)) {}
    void draw() const override {
        obj.draw();
    }
    
    std::unique_ptr<Concept>
    clone() const override {
        return std::make_unique<Model>(obj);
    }
};
            

// Copy constructor & assignment
Drawable(const Drawable& other)
    : self(other.self->clone()) {}

Drawable& operator=(const Drawable& other)
{
    self = other.self->clone();
    return *this;
}

// We force move semantic
Drawable(Drawable&&) noexcept = default;
Drawable& operator=(Drawable&&) noexcept
    = default;

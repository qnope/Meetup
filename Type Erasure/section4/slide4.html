Step 3 — Create Model<T>

A template that wraps any type with a draw() method.


class Drawable {
    struct Concept {
        virtual ~Concept() = default;
        virtual void draw() const = 0;
    };

    template <typename T>
    struct Model : Concept {
        T obj;

        Model(T obj) : obj(std::move(obj)) {}

        void draw() const override {
            obj.draw();
        }
    };

    std::unique_ptr<Concept> self;
};

Model<T> bridges any concrete type to the internal Concept.

Step 1 — Classic Inheritance


struct Shape {
    virtual ~Shape() = default;
    virtual void draw() const = 0;
};

struct Circle : Shape {
    double radius;
    void draw() const override {
        std::cout << "Circle(" << radius << ")\n";
    }
};

struct Square : Shape {
    double side;
    void draw() const override {
        std::cout << "Square(" << side << ")\n";
    }
};
    

std::vector<std::unique_ptr<Shape>> shapes;
shapes.push_back(std::make_unique<Circle>(Circle{2.0}));
shapes.push_back(std::make_unique<Square>(Square{3.0}));

Problem: intrusive, requires pointers, no value semantics.

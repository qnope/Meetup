Step 4 bis — Heterogeneous Collection


// No inheritance needed!
struct Circle {
    double radius;
    void draw() const { std::cout << "Circle(" << radius << ")\n"; }
};

struct Square {
    double side;
    void draw() const { std::cout << "Square(" << side << ")\n"; }
};
    

// Value semantics, no pointers in user code!
std::vector<Drawable> shapes;
shapes.push_back(Circle{2.0});
shapes.push_back(Square{3.0});

for (auto& s : shapes) {
    s.draw();
}

No base class, no pointers, no variant — just values.

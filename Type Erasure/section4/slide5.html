Step 4 — Template Constructor + Public API


class Drawable {
    struct Concept { /* ... */ };

    template <typename T>
    struct Model : Concept { /* ... */ };

    std::unique_ptr<Concept> self;

public:
    template <typename T>
    Drawable(T obj)
        : self(std::make_unique<Model<T>>(std::move(obj))) {}

    void draw() const {
        self->draw();
    }
};

The constructor deduces T automatically — any type with draw() just works.

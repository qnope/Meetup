std::function


// Lambda
std::function<int(int, int)> add = [](int a, int b) { return a + b; };

// Function pointer
int multiply(int a, int b) { return a * b; }
std::function<int(int, int)> mul = multiply;

// Functor
struct Divider {
    int operator()(int a, int b) { return a / b; }
};
std::function<int(int, int)> div = Divider{};

One type, three different callables.

Internally: concept/model pattern + small buffer optimization.

The Concept / Model Pattern

Wrapper — public API, value semantics

Concept — private abstract interface

Model<T> — template that wraps any conforming type

class Drawable {
  // public draw()
  // stores unique_ptr<Concept>    

  struct Concept {
    virtual void draw() = 0;
  };

  template<typename T>
  struct Model : Concept {
    T obj;
    void draw() {
      obj.draw();
    }
  };
};

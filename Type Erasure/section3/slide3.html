std::any


std::any a = 42;
std::cout << std::any_cast<int>(a) << "\n";        // 42

a = std::string("hello");
std::cout << std::any_cast<std::string>(a) << "\n"; // hello

a = Circle{5.0};
std::any_cast<Circle>(a).draw();                     // Circle(5)
  • Stores any type with value semantics
  • any_cast throws bad_any_cast on type mismatch
  • No interface— you must know the type to extract it

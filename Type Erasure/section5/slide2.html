std::polymorphic<Shape>

Value semantics + polymorphism, built into the standard. (P3019)


struct Shape {
    virtual ~Shape() = default;
    virtual void draw() const = 0;
};

struct Circle : Shape {
    double radius;
    void draw() const override {
        std::cout << "Circle(" << radius << ")\n";
    }
};
    

std::vector<std::polymorphic<Shape>> shapes;
shapes.push_back(Circle{2.0});
shapes.push_back(Square{3.0});

auto copy = shapes; // deep copy, just works!

Copyable, movable, no unique_ptr, no slicing.

Our Drawable vs std::polymorphic

Our Drawable (~30 lines)


class Drawable {
    struct Concept { /* ... */ };
    template<typename T>
    struct Model : Concept { /* ... */ };
    std::unique_ptr<Concept> self;
public:
    template<typename T>
    Drawable(T obj);
    void draw() const;
    // + copy/move
};

Duck-typed: any type with draw()

std::polymorphic


std::polymorphic<Shape> shape = Circle{2.0};

// That's it.
// Deep copy, value semantics,
// polymorphism — all built-in.

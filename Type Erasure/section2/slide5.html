std::variant


using Shape = std::variant<Circle, Square>;

std::vector<Shape> shapes;
shapes.push_back(Circle{2.0});
shapes.push_back(Square{3.0});

for (auto& s : shapes) {
    std::visit([](auto& shape) { shape.draw(); }, s);
}
  • Closed set — every type must be listed at compile time
  • Visitor is verbose — gets unwieldy with many types
  • Adding a new type requires modifying the variant definition
